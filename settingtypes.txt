# Maximum distance between smartshop and external storage
smartshop.max_wifi_distance (Smartshop max storage distance) int 30 0 113512

# Number of seconds to link external storage
smartshop.wifi_link_time (Smartshop storage link time) int 30 0 3600

# Enable currency changing (experimental feature).
smartshop.change_currency (Enable currency changing) bool true

# If set to false, this will disable giving refunds for the "give" and "pay" slots in a shop when upgrading from the main fork.
smartshop.enable_refund (Enable refunds on upgrading from the main fork) bool true

# A privilege which allows the player to run admin shops
smartshop.admin_shop_privs (Privs for admin shops) string smartshop_admin
